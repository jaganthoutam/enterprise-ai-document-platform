# Build stage
FROM node:18-alpine AS build

# Set working directory
WORKDIR /app

# Copy package files
COPY package.json package-lock.json ./

# Install dependencies
RUN npm ci

# Copy all files
COPY . .

# Set environment variables for production build
ARG REACT_APP_API_ENDPOINT
ARG REACT_APP_USER_POOL_ID
ARG REACT_APP_APP_CLIENT_ID
ARG REACT_APP_REGION
ARG REACT_APP_ENV

ENV REACT_APP_API_ENDPOINT=$REACT_APP_API_ENDPOINT
ENV REACT_APP_USER_POOL_ID=$REACT_APP_USER_POOL_ID
ENV REACT_APP_APP_CLIENT_ID=$REACT_APP_APP_CLIENT_ID
ENV REACT_APP_REGION=$REACT_APP_REGION
ENV REACT_APP_ENV=$REACT_APP_ENV

# Build the application
RUN npm run build

# Production stage
FROM nginx:alpine

# Copy built files from build stage
COPY --from=build /app/build /usr/share/nginx/html

# Copy custom nginx configuration
COPY nginx/nginx.conf /etc/nginx/conf.d/default.conf

# Expose port 80
EXPOSE 80

# Start Nginx server
CMD ["nginx", "-g", "daemon off;"]
